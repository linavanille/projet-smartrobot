        \Tq{etat $\neq$ ESorti \textbf{et} prochaineAction $\neq$ .}{
            \Suivant{etat}{
                \Cas{EAvancer}{avancer (etat)}
                \Cas{EIntersection}{tourner (etat)}

                }
            numOrdre $\leftarrow$ numOrdre + 1\\
            prochaineAction $\leftarrow$ listeOrdre[numOrdre]\\
        }
    \end{algorithm}

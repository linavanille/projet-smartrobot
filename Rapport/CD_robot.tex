\section*{Procédures}

\noindent\textbf{procedure} ROBOT\_avancer (\textbf{E/S:} etat: \textbf{ROBOT\_EtatDAvancement})\\
\textbf{début}
    \begin{algorithm}
        \Tq{(\textbf{non} SENS\_instersection) \textbf{et} (etat $\neq$ Urgence)}{
            \eSi{USON\_estDevantUnObstacle (ECHO, TRIG)}{etat $\leftarrow$ Urgence}
            {MTR\_avancer (MOTEUR\_IN1, MOTEUR\_IN2, MOTEUR\_IN3, MOTEUR\_IN4, PWM\_EN1, PWM\_EN2)\\
                \uSi{$SENS\_redresserADroite$}{
                    MTR\_redresser (PWM\_EN2, PWM\_EN1)\\
                    \Repeter{digitalRead (LINE\_SENSOR\_FRONT) = 1}{
                        \Si{USON\_estDevantUnObstacle (ECHO, TRIG)}{ROBOT\_urgence ()}
                    }
                    MTR\_redresser (PWM\_EN1, PWM\_EN2)\\
                    attendre ()\\
                    MTR\_avancer (MOTEUR\_IN1, MOTEUR\_IN2, MOTEUR\_IN3, MOTEUR\_IN4, PWM\_EN1, PWM\_EN2)
                }
                \SinonSi{$SENS\_redresserAGauche$}{
                    MTR\_redresser (PWM\_EN1, PWM\_EN2)\\
                    \Repeter{digitalRead (LINE\_SENSOR\_FRONT) = 1}{
                        \Si{USON\_estDevantUnObstacle (ECHO, TRIG)}{ROBOT\_urgence ()}
                    }
                    MTR\_redresser (PWM\_EN2, PWM\_EN1)\\
                    attendre ()\\
                    MTR\_avancer (MOTEUR\_IN1, MOTEUR\_IN2, MOTEUR\_IN3, MOTEUR\_IN4, PWM\_EN1, PWM\_EN2)
                }
            etat $\leftarrow$ Intersection
            }
        }
    \end{algorithm}

\noindent\textbf{fin}

\newpage\noindent\textbf{procedure} ROBOT\_intersection (\textbf{E/S:} etat: \textbf{ROBOT\_EtatDAvancement}, \textbf{E:} prochaineAction: \textbf{ORD\_Ordre})\\
\textbf{début}

    \begin{algorithm}[H]
        \Suivant{prochainVirage}{
            \uCas{TD}{MTR\_tournerDroite (MOTEUR\_IN1, MOTEUR\_IN2, MOTEUR\_IN3, MOTEUR\_IN4)\\attendre()\\ etat $\leftarrow$ Avancer}
            \uCas{TG}{MTR\_tournerGauche (MOTEUR\_IN1, MOTEUR\_IN2, MOTEUR\_IN3, MOTEUR\_IN4)\\attendre()\\ etat $\leftarrow$ Avancer}
            \uCas{AV}{etat $\leftarrow$ Avancer}
            \Autre{etat = Sorti}
            }
        
    \end{algorithm}

\noindent\textbf{fin}\\

\noindent\textbf{procedure} ROBOT\_urgence ()\\
\textbf{début}

    \begin{algorithm}[H]
        \Tq{USON\_estDevantUnObstacle (ECHO, TRIG)}{
            MTR\_arreter (MOTEUR\_IN1, MOTEUR\_IN2, MOTEUR\_IN3, MOTEUR\_IN4)
        }
        attendre (1000)        
    \end{algorithm}

\noindent\textbf{fin}\\

\noindent\textbf{procedure} ROBOT\_evolutionRobot (\textbf{E:} listeDOrdres: \textbf{ORD\_ListeDOrdres})

\begin{tabular}{ll}
    \textbf{Déclaration:} & prochaineAction: \textbf{ORD\_Ordre}\\
     & etat: \textbf{ROBOT\_EtatDAvancement}\\
     & numOrdre: \textbf{Naturel Non Nul}
\end{tabular}

\noindent\textbf{début}

    \begin{algorithm}[H]
        numOrdre $\leftarrow$ 1\\
        etat $\leftarrow$ Avancer\\
        prochaineAction $\leftarrow$ listeDOrdre[numOrdre]\\
        \Tq{etat $\neq$ Sorti \textbf{et} prochaineAction $\neq$.}{
            \Suivant{etat}{
                \Cas{Avancer}{ROBOT\_avancer (etat)}
                \Cas{Intersection}{tourner (etat)}
                \Cas{Urgence}{ROBOT\_urgence ()}
                }
            numOrdre $\leftarrow$ numOrdre + 1\\
            prochaineAction $\leftarrow$ listeOrdre[numOrdre]\\
        }
    \end{algorithm}

    \noindent\textbf{fin}

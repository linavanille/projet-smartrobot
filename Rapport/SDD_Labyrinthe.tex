\section*{Structures de Données}

\begin{algorithme}
    \type{Case}{\tableau{1..4}{de}{\naturel}}

    \begin{enregistrement}{Labyrinthe}
        \champEnregistrement{lesConnections}{\tableau{1..MAX}{de}{Case}}
        \champEnregistrement{entree}{\naturelNonNul}
        \champEnregistrement{sortie}{\naturelNonNul}
        \champEnregistrement{largeur}{\naturelNonNul}
    \end{enregistrement}
    \end{algorithme}

\section*{Conception Préliminaire Labyrinthe}

\begin{algorithme}
    \signatureFonction{LAB\_creerLabyrinthe}
    {l: \naturelNonNul, entree, sortie: Case}{Labyrinthe}
    {}

    \signatureFonction{LAB\_obtenirLargeur}
    {lab : Labyrinthe}{\naturelNonNul}
    {}

    \signatureFonction{LAB\_obtenirCasesAccessibles}
    {lab : Labyrinthe, uneCase : Case}{\tableau{1..4}{de}{Case}}
    {}

    \signatureFonction{LAB\_obtenirEntree}
    {lab : Labyrinthe}{\naturelNonNul}
    {}

    \signatureFonction{LAB\_obtenirSortie}
    {lab : Labyrinthe}{\naturelNonNul}
    {}

    \signatureFonction{LAB\_iemeCase}
    {lab : Labyrinthe, iCase: \naturelNonNul}{Case}
    {}
\end{algorithme}

\section*{Conception Détaillée Labyrinthe}

\begin{algorithme}

    \fonction{LAB\_creerLabyrinthe}{l : \naturelNonNul, entree, sortie: Case}{Labyrinthe}
    {}{}
    {
        \affecter{lab}{\tableau{1..l*l}{de}{Case}}
        \affecter{lab.entree}{entree}
        \affecter{lab.sortie}{sortie}
        \affecter{lab.largeur}{largeur}
    }
    
\end{algorithme}

\begin{algorithme}

    \fonction{LAB\_obtenirLargeur}{lab : Labyrinthe}{\naturelNonNul}
    {}{}
    {
        \retourner{lab.largeur}
    }
    
\end{algorithme}

\begin{algorithme}

    \fonction{LAB\_obtenirEntree}{lab : Labyrinthe}{\naturelNonNul}
    {}{}
    {
        \retourner{lab.entree}
    }
    
\end{algorithme}

\begin{algorithme}

    \fonction{LAB\_obtenirSortie}{lab : Labyrinthe}{\naturelNonNul}
    {}{}
    {
        \retourner{lab.sortie}
    }
    
\end{algorithme}

\begin{algorithme}

    \fonction{LAB\_obtenirCasesAccessibles}{lab : Labyrinthe, iCase : Case}{Case}
    {}{res : Case}
    {
        \affecter{res}{LAB\_iemeCase(lab, iCase)}
        \retourner{res}
    }
    
\end{algorithme}

\begin{algorithme}

    \fonction{LAB\_iemeCase}{lab : Labyrinthe, iCase : Case}{Case}
    {}{res : Case}
    {
        \affecter{res}{lab[iCase]}
        \retourner{res}
    }
    
\end{algorithme}

\section*{Conception Détaillée Case}

\begin{algorithme}

    \fonction{CASE\_creerCase}{n: \naturelNonNul}{Case\_Case}
    {}{c: CASE\_Case} 
    {
        \affecter{c.Numero}{n}
        \retourner{c}
    }   

\end{algorithme}
\begin{algorithme}

    \fonction{CASE\_obtenirNumeroCase}{CASE : CASE\_Case}{\naturelNonNul}
    {}{}
    {

    }    

\end{algorithme}

    \fonction{CASE\_obtenirHaut}{CASE : CASE\_Case}{\naturelNonNul}

\begin{algorithme}   

    \fonction{CASE\_obtenirGauche}{CASE : CASE\_Case}{\naturelNonNul}

\begin{algorithme}

    \fonction{CASE\_obtenirBas}{CASE : CASE\_Case}{\naturelNonNul}

    {}{}

        \retourner{CASE.Droite}
    }


    \procedure{CASE\_fixerHaut}{\paramEntreeSortie{c : CASE\_Case} \paramEntree{num : \naturelNonNul}}

    \fonction{CASE\_obtenirCasesAccessibles} {c : CASE\_Case}{\tableau{1..4}{de}{\naturelNonNul}}

    }
